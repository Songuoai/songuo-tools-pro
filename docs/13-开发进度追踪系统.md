## 🔧 进度保存机制

### 1. 自动保存脚本

```javascript
// scripts/save-progress.js
const fs = require('fs');
const path = require('path');

class ProgressTracker {
  constructor() {
    this.progressFile = path.join(__dirname, '../docs/progress/progress.json');
    this.historyDir = path.join(__dirname, '../docs/progress/progress-history');
    this.progress = this.loadProgress();
  }

  loadProgress() {
    try {
      const data = fs.readFileSync(this.progressFile, 'utf-8');
      return JSON.parse(data);
    } catch (error) {
      // 文件不存在，创建初始进度
      return this.createInitialProgress();
    }
  }

  createInitialProgress() {
    const initialProgress = {
      version: '1.0.0',
      lastUpdated: new Date().toISOString(),
      phase: 'Phase 1: 学习 + 初始化',
      overallProgress: {
        percentage: 0,
        startDate: '2026-03-24',
        targetDate: '2026-05-19',
        currentWeek: 1,
        totalWeeks: 12
      },
      learning: {
        currentTopic: '',
        completedTopics: [],
        nextTopic: '',
        learningLogs: []
      },
      development: {
        completedFeatures: [],
        inProgressFeatures: [],
        pendingFeatures: []
      },
      issues: {
        open: [],
        resolved: []
      },
      checkpoints: {
        lastCheckpoint: new Date().toISOString(),
        nextCheckpoint: new Date(Date.now() + 7 * 24 * 60 * 60 * 1000).toISOString(),
        checkpointProgress: ''
      }
    };

    this.saveProgress(initialProgress);
    return initialProgress;
  }

  saveProgress(progress = this.progress) {
    // 更新最后修改时间
    progress.lastUpdated = new Date().toISOString();

    // 保存到进度文件
    fs.writeFileSync(
      this.progressFile,
      JSON.stringify(progress, null, 2),
      'utf-8'
    );

    // 每日备份
    this.backupDaily();

    console.log('✅ 进度已保存:', progress.lastUpdated);
  }

  backupDaily() {
    const today = new Date().toISOString().split('T')[0];
    const backupPath = path.join(this.historyDir, `${today}.json`);

    // 确保目录存在
    if (!fs.existsSync(this.historyDir)) {
      fs.mkdirSync(this.historyDir, { recursive: true });
    }

    // 备份
    fs.writeFileSync(
      backupPath,
      JSON.stringify(this.progress, null, 2),
      'utf-8'
    );

    console.log('💾 已备份到:', backupPath);
  }

  // 添加已完成功能
  addCompletedFeature(feature) {
    const completed = {
      ...feature,
      completedAt: new Date().toISOString(),
      status: 'done',
      tested: false
    };

    this.progress.development.completedFeatures.push(completed);
    
    // 从进行中移除
    this.progress.development.inProgressFeatures = 
      this.progress.development.inProgressFeatures.filter(
        f => f.id !== feature.id
      );

    this.saveProgress();
  }

  // 更新进行中的功能
  updateInProgressFeature(featureId, updates) {
    const feature = this.progress.development.inProgressFeatures.find(
      f => f.id === featureId
    );

    if (feature) {
      Object.assign(feature, updates);
      this.saveProgress();
    }
  }

  // 添加问题
  addIssue(issue) {
    this.progress.issues.open.push({
      ...issue,
      createdAt: new Date().toISOString(),
      status: 'open'
    });
    this.saveProgress();
  }

  // 解决问题
  resolveIssue(issueId, solution) {
    const issue = this.progress.issues.open.find(i => i.id === issueId);
    if (issue) {
      issue.status = 'resolved';
      issue.resolvedAt = new Date().toISOString();
      issue.solution = solution;
      
      this.progress.issues.resolved.push(issue);
      this.progress.issues.open = 
        this.progress.issues.open.filter(i => i.id !== issueId);
      
      this.saveProgress();
    }
  }

  // 生成进度报告
  generateReport() {
    const report = {
      date: new Date().toISOString().split('T')[0],
      phase: this.progress.phase,
      overallProgress: this.progress.overallProgress.percentage,
      completedFeatures: this.progress.development.completedFeatures.length,
      inProgressFeatures: this.progress.development.inProgressFeatures.length,
      openIssues: this.progress.issues.open.length,
      currentLearning: this.progress.learning.currentTopic,
      lastUpdated: this.progress.lastUpdated
    };

    return report;
  }
}

module.exports = ProgressTracker;
```

### 3. 自动保存触发器

```javascript
// 在关键操作后自动保存进度

// 1. 完成一个功能后
function onFeatureComplete(featureId) {
  tracker.addCompletedFeature({
    id: featureId,
    name: '功能名称',
    category: '功能分类'
  });
}

// 2. 遇到问题时
function onIssueFound(issue) {
  tracker.addIssue({
    id: `issue-${Date.now()}`,
    title: '问题描述',
    severity: 'high|medium|low'
  });
}

// 3. 每日结束时
function onDayEnd() {
  tracker.backupDaily();
  tracker.generateReport();
}

// 4. 学习完成后
function onLearningComplete(topic) {
  tracker.progress.learning.completedTopics.push(topic);
  tracker.saveProgress();
}
```

## 💾 灾难恢复

### 进度恢复流程

```
1. 检查最新备份
   - 查看 docs/progress/progress.json
   - 查看 docs/progress/progress-history/ 最新文件

2. 恢复进度
   - 复制备份文件到 progress.json
   - 运行进度验证脚本

3. 验证恢复
   - 检查功能列表
   - 检查学习记录
   - 检查问题追踪

4. 继续开发
```

### 备份策略

```
- 实时保存：每次进度更新
- 每日备份：每天 23:00
- 每周备份：每周日单独备份
- 里程碑备份：每个检查点备份
```
