function getGreeting() {
  const hour = new Date().getHours()
  
  if (hour >= 5 && hour < 12) {
    return { text: '早上好', emoji: '🌅', tip: '今天也要高效工作哦！' }
  } else if (hour >= 12 && hour < 14) {
    return { text: '中午好', emoji: '☀️', tip: '午休时间，别忘了休息~' }
  } else if (hour >= 14 && hour < 18) {
    return { text: '下午好', emoji: '🌆', tip: '继续加油！' }
  } else if (hour >= 18 && hour < 23) {
    return { text: '晚上好', emoji: '🌙', tip: '注意劳逸结合~' }
  } else {
    return { text: '夜深了', emoji: '🌃', tip: '这么晚还在工作？注意休息！' }
  }
}
