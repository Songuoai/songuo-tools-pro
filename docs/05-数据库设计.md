## 📋 表结构详情

### 1. categories（分类表）

```sql
CREATE TABLE categories (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  name text NOT NULL,                    -- 分类名称
  slug text UNIQUE NOT NULL,             -- URL 友好名称
  icon text,                             -- 图标（emoji 或 SVG）
  description text,                      -- 分类描述
  parent_id uuid REFERENCES categories(id), -- 父分类 ID
  sort_order integer DEFAULT 0,          -- 排序
  is_active boolean DEFAULT true,        -- 是否启用
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);

-- 索引
CREATE INDEX idx_categories_parent ON categories(parent_id);
CREATE INDEX idx_categories_slug ON categories(slug);
CREATE INDEX idx_categories_active ON categories(is_active);
```

### 2. tools（工具表）

```sql
CREATE TABLE tools (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  name text NOT NULL,                    -- 工具名称
  slug text UNIQUE NOT NULL,             -- URL 友好名称
  description text NOT NULL,             -- 简短描述（200 字内）
  content text,                          -- 详细介绍（Markdown）
  website_url text NOT NULL,             -- 官网链接
  logo_url text,                         -- Logo 图片 URL
  screenshots text[] DEFAULT '{}',       -- 截图数组
  category_id uuid REFERENCES categories(id), -- 主分类
  subcategory_id uuid REFERENCES categories(id), -- 子分类
  tags text[] DEFAULT '{}',              -- 标签数组
  rating_avg numeric(3,2) DEFAULT 0,     -- 平均评分
  rating_count integer DEFAULT 0,        -- 评分数量
  click_count integer DEFAULT 0,         -- 点击次数
  favorite_count integer DEFAULT 0,      -- 收藏次数
  status text DEFAULT 'draft',           -- draft, published, archived
  is_featured boolean DEFAULT false,     -- 是否精选
  featured_order integer DEFAULT 0,      -- 精选排序
  seo_title text,                        -- SEO 标题
  seo_description text,                  -- SEO 描述
  seo_keywords text[],                   -- SEO 关键词
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now(),
  published_at timestamptz
);

-- 索引
CREATE INDEX idx_tools_slug ON tools(slug);
CREATE INDEX idx_tools_category ON tools(category_id);
CREATE INDEX idx_tools_subcategory ON tools(subcategory_id);
CREATE INDEX idx_tools_tags ON tools USING GIN(tags);
CREATE INDEX idx_tools_status ON tools(status);
CREATE INDEX idx_tools_featured ON tools(is_featured);
CREATE INDEX idx_tools_rating ON tools(rating_avg DESC);
CREATE INDEX idx_tools_clicks ON tools(click_count DESC);
CREATE INDEX idx_tools_published ON tools(published_at DESC);

-- 全文搜索索引
CREATE INDEX idx_tools_search ON tools USING GIN(
  to_tsvector('simple', name || ' ' || description || ' ' || array_to_string(tags, ' '))
);
```

### 3. users（用户表 - 扩展 Supabase Auth）

```sql
CREATE TABLE users (
  id uuid PRIMARY KEY REFERENCES auth.users(id),
  email text UNIQUE NOT NULL,
  nickname text,
  avatar_url text,
  bio text,                            -- 个人简介
  role text DEFAULT 'user',            -- user, admin, super_admin
  is_verified boolean DEFAULT false,   -- 邮箱验证
  last_login_at timestamptz,
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);

-- 索引
CREATE INDEX idx_users_email ON users(email);
CREATE INDEX idx_users_role ON users(role);

-- 触发器：自动创建用户记录
CREATE OR REPLACE FUNCTION handle_new_user()
RETURNS trigger AS $$
BEGIN
  INSERT INTO public.users (id, email, nickname)
  VALUES (new.id, new.email, split_part(new.email, '@', 1));
  RETURN new;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE FUNCTION handle_new_user();
```

### 5. comments（评论表）

```sql
CREATE TABLE comments (

  is_helpful boolean DEFAULT false,      -- 是否有用
  helpful_count integer DEFAULT 0,       -- 有用计数
  created_at timestamptz DEFAULT now(),
  updated_at timestamptz DEFAULT now()
);

-- 索引
CREATE INDEX idx_comments_tool ON comments(tool_id);
CREATE INDEX idx_comments_user ON comments(user_id);
CREATE INDEX idx_comments_parent ON comments(parent_id);
CREATE INDEX idx_comments_status ON comments(status);
```

### 8. analytics（统计表）

```sql
CREATE TABLE analytics (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
```

## 🔐 Row Level Security (RLS) 策略

### tools 表
```sql
ALTER TABLE tools ENABLE ROW LEVEL SECURITY;

-- 所有人可读已发布的工具
CREATE POLICY "Public tools are viewable by everyone"
  ON tools FOR SELECT
  USING (status = 'published');

-- 仅管理员可插入/更新/删除
CREATE POLICY "Admins can manage tools"
  ON tools FOR ALL
  USING (
    EXISTS (
      SELECT 1 FROM users
      WHERE users.id = auth.uid()
      AND users.role IN ('admin', 'super_admin')
    )
  );
```

### favorites 表
```sql
ALTER TABLE favorites ENABLE ROW LEVEL SECURITY;

-- 用户可查看自己的收藏
CREATE POLICY "Users can view own favorites"
  ON favorites FOR SELECT
  USING (auth.uid() = user_id);

-- 用户可添加自己的收藏
CREATE POLICY "Users can insert own favorites"
  ON favorites FOR INSERT
  WITH CHECK (auth.uid() = user_id);

-- 用户可删除自己的收藏
CREATE POLICY "Users can delete own favorites"
  ON favorites FOR DELETE
  USING (auth.uid() = user_id);
```

### comments 表
```sql
ALTER TABLE comments ENABLE ROW LEVEL SECURITY;

-- 所有人可读已审核的评论
CREATE POLICY "Public comments are viewable"
  ON comments FOR SELECT
  USING (status = 'approved');

-- 登录用户可发表评论
CREATE POLICY "Users can create comments"
  ON comments FOR INSERT
  WITH CHECK (auth.uid() = user_id);

-- 用户可修改自己的评论
CREATE POLICY "Users can update own comments"
  ON comments FOR UPDATE
  USING (auth.uid() = user_id);

-- 管理员可管理所有评论
CREATE POLICY "Admins can manage comments"
  ON comments FOR ALL
  USING (
    EXISTS (
      SELECT 1 FROM users
      WHERE users.id = auth.uid()
      AND users.role IN ('admin', 'super_admin')
    )
  );
```

## ⚡ 数据库函数

### 更新工具评分
```sql
CREATE OR REPLACE FUNCTION update_tool_rating()
RETURNS trigger AS $$
BEGIN
  IF TG_OP = 'INSERT' OR TG_OP = 'UPDATE' THEN
    UPDATE tools
    SET rating_avg = (
          SELECT COALESCE(AVG(rating), 0)
          FROM comments
          WHERE tool_id = NEW.tool_id
          AND status = 'approved'
          AND rating IS NOT NULL
        ),
        rating_count = (
          SELECT COUNT(*)
          FROM comments
          WHERE tool_id = NEW.tool_id
          AND status = 'approved'
          AND rating IS NOT NULL
        )
    WHERE id = NEW.tool_id;
  END IF;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER on_comment_rating_change
  AFTER INSERT OR UPDATE ON comments
  FOR EACH ROW EXECUTE FUNCTION update_tool_rating();
```

### 记录工具点击
```sql
CREATE OR REPLACE FUNCTION record_tool_click()
RETURNS trigger AS $$
BEGIN
  INSERT INTO analytics (event_type, tool_id, session_id, metadata)
  VALUES ('click', NEW.id, current_setting('app.session_id', true), '{}');
  
  UPDATE tools
  SET click_count = click_count + 1
  WHERE id = NEW.id;
  
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;
```

## 📊 视图（Views）

### 热门工具视图
```sql
CREATE VIEW v_popular_tools AS
SELECT 
  t.*,
  c.name as category_name,
  c.slug as category_slug
FROM tools t
LEFT JOIN categories c ON t.category_id = c.id
WHERE t.status = 'published'
ORDER BY 
  t.click_count * 0.4 + 
  t.favorite_count * 0.3 + 
  t.rating_avg * 10 * 0.3 DESC
LIMIT 100;
```

### 最新工具视图
```sql
CREATE VIEW v_latest_tools AS
SELECT 
  t.*,
  c.name as category_name
FROM tools t
LEFT JOIN categories c ON t.category_id = c.id
WHERE t.status = 'published'
ORDER BY t.published_at DESC
LIMIT 100;
```

## 🔍 搜索优化

### 工具搜索函数
```sql
CREATE OR REPLACE FUNCTION search_tools(
  query_text text,
  category_filter uuid DEFAULT NULL,
  limit_count integer DEFAULT 20
)
RETURNS TABLE (
  id uuid,
  name text,
  description text,
  slug text,
  rating_avg numeric,
  click_count integer,
  relevance float
) AS $$
BEGIN
  RETURN QUERY
  SELECT 
    t.id,
    t.name,
    t.description,
    t.slug,
    t.rating_avg,
    t.click_count,
    ts_rank(
      to_tsvector('simple', t.name || ' ' || t.description || ' ' || array_to_string(t.tags, ' ')),
      to_tsquery('simple', query_text)
    ) as relevance
  FROM tools t
  WHERE t.status = 'published'
    AND (category_filter IS NULL OR t.category_id = category_filter)
    AND to_tsvector('simple', t.name || ' ' || t.description || ' ' || array_to_string(t.tags, ' '))
        @@ to_tsquery('simple', query_text)
  ORDER BY relevance DESC, t.click_count DESC
  LIMIT limit_count;
END;
$$ LANGUAGE plpgsql;
```
