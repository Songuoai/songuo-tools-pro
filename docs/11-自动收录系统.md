### 信息提取技术

#### 1. Meta 标签提取
```javascript
async function extractMetaInfo(url) {
  const response = await fetch(url)
  const html = await response.text()
  
  return {
    // 标题
    title: extractOgTag(html, 'og:title') || 
           extractTag(html, 'title'),
    
    // 描述
    description: extractOgTag(html, 'og:description') || 
                 extractTag(html, 'meta[name="description"]'),
    
    // Logo
    logo: extractOgTag(html, 'og:image') || 
          extractFavicon(url),
    
    // 关键词
    keywords: extractTag(html, 'meta[name="keywords"]')?.split(',')
  }
}
```

async function autoScreenshot(url) {
  // 使用 Puppeteer 或 Playwright
  const browser = await puppeteer.launch()
  const page = await browser.newPage()
  
  await page.goto(url, { waitUntil: 'networkidle2' })
  
  // 截取首页截图
  const screenshot = await page.screenshot({
    fullPage: false,
    clip: { x: 0, y: 0, width: 1200, height: 800 }
  })
  
  // 上传到存储

### 重复检测

#### 去重算法
```javascript
async function checkDuplicate(newTool) {
  // 1. URL 去重
  const existingByUrl = await findToolByUrl(newTool.url)
  if (existingByUrl) {
    return { isDuplicate: true, reason: 'URL 已存在' }
  }
  
  // 2. 名称去重（模糊匹配）
  const existingByName = await findToolByName(newTool.name)
  if (existingByName) {
    return { isDuplicate: true, reason: '名称相似' }
  }
  
  // 3. 内容相似度检测
  const similarTools = await findSimilarTools(newTool.description)
  if (similarTools.length > 0) {
    return { 
      isDuplicate: false, 
      similar: similarTools,
      warning: '发现相似工具，请确认是否重复'
    }
  }
  
  return { isDuplicate: false }
}
```
